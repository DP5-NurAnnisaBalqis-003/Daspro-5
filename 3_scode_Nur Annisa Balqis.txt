program kuadran;

uses crt;

var
        x,y,titik_koordinat:integer;
begin
        clrscr;
        write('Masukkan titik x         : ');
        readln(x);
        write('Masukkan titik y         : ');
        readln(y);
        if (x=0) and (y=0) then
        begin
                titik_koordinat:=0;
                end
        else if (x>0) and (y>0) then
        begin
                titik_koordinat:=1;
                end
        else if (x<0) and (y>0) then
        begin
                titik_koordinat:=2;
                end
        else if (x<0) and (y<0) then
        begin
                titik_koordinat:=3;
                end
        else if (x>0) and (y<0) then
        begin
                titik_koordinat:=4;
                end;
        writeln('Titik Anda berada di kuadran ', titik_koordinat );
        readln;
end.
